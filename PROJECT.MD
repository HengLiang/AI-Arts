<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=1920, height=1080">
    <title>赛博安全屏保 - 零知之证 | Cyber Security Screensaver</title>

            width: 1920px;
            height: 1080px;
        }

            position: fixed;
            bottom: 10px;
            right: 10px;
            color: #00FF00;
            font-family: monospace;
            font-size: 12px;
            opacity: 0.5;
            pointer-events: none;
        }
    </style>
</head>
<body>
    <canvas id="matrix"></canvas>
    <div class="info">Zero Knowledge Proof | SHA-256 | Merkle Tree</div>
    <script>
        const canvas = document.getElementById('matrix');
        const ctx = canvas.getContext('2d');
        
        canvas.width = 1920;
        canvas.height = 1080;
        
        // 字符源：100%来自诗歌《零知之证》
        const poemChars = '零知之证SHA256在黑暗中旋转椭圆曲线的私钥沉睡MerkleTree的叶子飘落每一片都是不可逆的誓言我知秘密但不说破ZeroKnowledge的优雅舞蹈验证者看见真相的轮廓却永远触不到核心脉搏数字签名跨越时空公钥锁住私钥开启量子威胁在远方等待后密码学正在酝酿黎明区块链上节点低语共识达成无需中介分布式账本记录永恒密码学是数字世界的基石加密不是隐藏是保护隐私不是秘密是权利在0和1的宇宙里我们是自己密钥的守护者';
        
        const characters = poemChars.split('');
        const fontSize = 16;
        const columns = canvas.width / fontSize;
        
        // 初始化掉落位置（随机起始，形成错落感）
        const drops = [];
        for(let i = 0; i < columns; i++) {
            drops[i] = Math.random() * -100;
        }
        
        // 每列独立速度（增加层次感）
        const speeds = [];
        for(let i = 0; i < columns; i++) {
            speeds[i] = 0.5 + Math.random() * 2;
        }
        
        // 字符亮度衰减（模拟Matrix效果）
        const brightness = [];
        for(let i = 0; i < columns; i++) {
            brightness[i] = 0.8 + Math.random() * 0.2;
        }
        
        function draw() {
            // 半透明黑色覆盖（产生拖尾效果）
            ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            for(let i = 0; i < drops.length; i++) {
                // 随机选择诗歌字符
                const char = characters[Math.floor(Math.random() * characters.length)];
                
                // 计算颜色（绿色渐变）
                const green = Math.floor(170 * brightness[i] + 85);
                ctx.fillStyle = `rgb(0, ${green}, 0)`;
                
                // 设置字体
                ctx.font = fontSize + 'px monospace';
                
                // 绘制字符
                ctx.fillText(char, i * fontSize, drops[i] * fontSize);
                
                // 重置到顶部（随机概率，形成不规则掉落）
                if(drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
                    drops[i] = 0;
                    // 重置时随机化速度和亮度
                    speeds[i] = 0.5 + Math.random() * 2;
                    brightness[i] = 0.8 + Math.random() * 0.2;
                }
                
                // 更新位置
                drops[i] += speeds[i];
            }
        }
        
        // 30 FPS 动画循环
        setInterval(draw, 33);
        
        // 点击暂停/继续
        let isPaused = false;
        document.addEventListener('click', () => {
            isPaused = !isPaused;
            if(!isPaused) setInterval(draw, 33);
        });
        
        // ESC退出提示
        document.addEventListener('keydown', (e) => {
            if(e.key === 'Escape') {
                alert('按浏览器返回键或关闭窗口退出屏保');
            }
        });
    </script>
</body>
</html>

const { app, BrowserWindow, screen } = require('electron');
